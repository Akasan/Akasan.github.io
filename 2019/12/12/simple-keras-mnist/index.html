KerasによるシンプルなMNISTのClassification

tl;dr

とにかくディープラーニングはじめた人が絶対やるであろうMNISTのClassificationをいまさらながらちゃんとやって記事にした

```python
import numpy as np
from keras.models import Sequential
from keras.layers import Dense, Activation, Dropout
from keras.datasets import mnist
from keras.utils import np_utils

CLASS_NUM = 10
EPOCHS = 10
BATCH_SIZE = 128
VERBOSE = True

def generate_data():
“”” generate dataset”””
(x_train, y_train), (x_test, y_test) = mnist.load_data()
x_train = x_train.astype(np.float32) / 255.0
x_test = x_test.astype(np.float32) / 255.0
x_train = x_train.reshape(-1, 28 * 28)
x_test = x_test.reshape(-1, 28 * 28) 

y_train = np_utils.to_categorical(y_train, CLASS_NUM)
y_test = np_utils.to_categorical(y_test, CLASS_NUM)

return (x_train, y_train), (x_test, y_test)

def generate_model(loss=”categorical_crossentropy”, optimizer=”adam”, metrics=[“accuracy”]):
“”” generate model

Arguments:
    loss {str} -- loss function
    optimizer {str} -- optimizer
    metrics {list(str)} -- metrics for calculating model's performance

Returns:
    {keras.models.Sequential} -- model
"""
model = Sequential()
model.add(Dense(128, input_shape=(28 * 28,)))
model.add(Activation("relu"))
model.add(Dense(256))
model.add(Activation("relu"))
model.add(Dense(CLASS_NUM))
model.add(Activation("softmax"))
model.compile(loss=loss, optimizer=optimizer, metrics=metrics)
return model

train, test = generate_data()
model = generate_model()
history = model.fit(train[0], train[1], epochs=EPOCHS, batch_size=BATCH_SIZE, verbose=VERBOSE)
score = model.evaluate(test[0], test[1], verbose=VERBOSE)
print(f”train loss: {score[0]:.3f}”)
print(f”train accuracy: {score[1]: .3f}”)
```
